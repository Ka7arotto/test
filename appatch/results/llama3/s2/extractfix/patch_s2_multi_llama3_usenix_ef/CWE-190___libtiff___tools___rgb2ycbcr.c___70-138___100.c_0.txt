main(int argc, char* argv[])

    int c;
    int width, height;
#if !HAVE_DECL_OPTARG
    extern int optind;
    extern char *optarg;
#endif

    while ((c = getopt(argc, argv, "i:o:w:h:")) != -1) {
        switch (c) {
            case 'i':
                in = TIFFOpen(optarg, "r");
                if (in == NULL) {
                    fprintf(stderr, "Error opening input file: %s\n", optarg);
                    return 1;
                }
                break;
            case 'o':
                out = TIFFOpen(optarg, "w");
                if (out == NULL) {
                    fprintf(stderr, "Error opening output file: %s\n", optarg);
                    return 1;
                }
                break;
            case 'w':
                width = atoi(optarg);
                break;
            case 'h':
                height = atoi(optarg);
                break;
            default:
                fprintf(stderr, "Usage: %s -i input.tif -o output.tif -w width -h height\n", argv[0]);
                return 1;
        }
    }

    // Resize TIFF image
    TIFFSetField(out, TIFFTAG_IMAGEWIDTH, width);
    TIFFSetField(out, TIFFTAG_IMAGELENGTH, height);
    TIFFWriteEncodedStrip(out, 0, TIFFGetStrip(in, 0), TIFFGetStripSize(in, 0));

    TIFFClose(in);
    TIFFClose(out);
    return 0;
}

int
main(int argc, char* argv[])
{
    TIFF *in, *out;
    int c;
    int x, y, width, height;
#if !HAVE_DECL_OPTARG
    extern int optind;
    extern char *optarg;
#endif

    while ((c = getopt(argc, argv, "i:o:x:y:w:h:")) != -1) {
        switch (c) {
            case 'i':
                in = TIFFOpen(optarg, "r");
                if (in == NULL) {
                    fprintf(stderr, "Error opening input file: %s\n", optarg);
                    return 1;
                }
                break;
            case 'o':
                out = TIFFOpen(optarg, "w");
                if (out == NULL) {
                    fprintf(stderr, "Error opening output file: %s\n", optarg);
                    return 1;
                }
                break;
            case 'x':
                x = atoi(optarg);
                break;
            case 'y':
                y = atoi(optarg);
                break;
            case 'w':
                width = atoi(optarg);
                break;
            case 'h':
                height = atoi(optarg);
                break;
            default:
                fprintf(stderr, "Usage: %s -i input.tif -o output.tif -x x -y y -w width -h height\n", argv[0]);
                return 1;
        }
    }

    // Crop TIFF image
    TIFFSetField(out, TIFFTAG_IMAGEWIDTH, width);
    TIFFSetField(out, TIFFTAG_IMAGELENGTH, height);
    TIFFSetField(out, TIFFTAG_XRESOLUTION, x);
    TIFFSetField(out, TIFFTAG_YRESOLUTION, y);
    TIFFWriteEncodedStrip(out, 0, TIFFGetStrip(in, 0), TIFFGetStripSize(in, 0));

    TIFFClose(in);
    TIFFClose(out);
    return 0;
}
